# 模拟call

Function.prototype.call2 = function(obj) {
                var args = [];//解决传入参数数量不固定问题，或用...运算符
                for (var i = 1; i < arguments.length; i++) {
                    args.push(i);
                }
                var obj = obj || window//如果call传入空值则指向window对象
                obj.fun = this//this代表调用call的函数，在目标对象中创建一个一样的函数
                var result = eval('obj.fun('+args+')');
                delete obj.fun;//删除函数
                if (result) return result;//如果函数有返回值则返回这个值
            }

# 模拟apply

Function.prototype.apply2 = function(obj,args) {
                var obj = obj || window
                obj.fun = this
                var result = eval('obj.fun('+args+')');
                delete obj.fun;
                if (result) return result;
            }

# 模拟bind

Function.prototype.bind2 = function(obj) {
    var obj = obj || window
    var that = this
    var args = Array.prototype.slice.call(arguments, 1);
    var bindfun = function() {
        var bindArgs = Array.prototype.slice.call(arguments);
        return that.apply(this instanceof that ? this : obj, args.concat(bindArgs))
    }
    var nfun = function() {};
    nfun.prototype = this.prototype;
    bindfun.prototype = new nfun();
    return bindfun;
}
