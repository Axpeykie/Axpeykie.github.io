# 第一版

利用Object.create()将实例对象的原型链接到构造函数的原型对象

function newanalog(cfun){
                var newObj = Object.create(cfun.prototype);
                var args= Array.prototype.slice.call(arguments,1);
                cfun.apply(newObj,args);
                return newObj
            }
//测试
function Person(name){
                this.name = name;
            }
            var person1 = newanalog(Person,"Tom");
            console.log(person1);//{name:"Tom"}
            console.log(person1.__proto__===Person.prototype);//true

# 第二版

如果构造函数的返回值不是基本类型，[new 构造函数]会直接返回这个值，而不会创建实例对象。
基于这个特点，我们可以做一个判断，让newanalog返回实例对象或构造函数返回值。

function newanalog(cfun) {
                var newObj = Object.create(cfun.prototype);
                var args = Array.prototype.slice.call(arguments, 1);
                var result = cfun.apply(newObj, args);
                if ((typeof result === 'object' && result !== null) || typeof result === 'function') {
                    return result
                }
                return newObj
            }
